def find_primes(n):
"""Return all primes up to n."""
sieve = [True] * (n + 1)
for i in range(2, int(n**0.5) + 1):
if sieve[i]:
sieve[i*i::i] = [False] * ((n - i*i) // i + 1)
return [i for i in range(2, n+1) if sieve[i]]
